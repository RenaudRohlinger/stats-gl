// Threejs example

import Stats from 'stats-gl'

const stats = new Stats({ trackGPU: true });
const renderer = new THREE.WebGPURenderer();
stats.init( renderer );
container.appendChild( stats.dom );

function animate() {
    requestAnimationFrame( animate );
    renderer.render( scene, camera );
    stats.update();
};
